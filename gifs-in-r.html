library(MASS)

proportional_permute <- function(x, prop) {
  indices <- seq(1:length(x))
  s_indices <- sample(indices)
  n_shuffle <- round(length(x) * prop)
  switch <- sample(indices)
  x[s_indices[1:n_shuffle]] <- x[switch[1:n_shuffle]]
  return(x)
}

}
